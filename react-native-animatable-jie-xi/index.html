// Transform an object to an array the way react native wants it for transform styles
// { a: x, b: y } => [{ a: x }, { b: y }]
function createKeyedArray(obj) {  
  return Object.keys(obj).map(key => {
    let keyed = {};
    keyed[key] = obj[key];
    return keyed;
  });
}
